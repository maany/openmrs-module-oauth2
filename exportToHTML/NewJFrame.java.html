148      private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed 
149          // TODO add your handling code here: 
150          String destMail = jTextField1.getText(); 
151          String sendMail = "sanattabrol@gmail.com"; 
152          String subjectMail = jTextField2.getText(); 
153          String contentMail = jTextArea1.getText(); 
154           
155          try { 
156   
157              Message message = new MimeMessage(SEND_SESSION); 
158              message.setFrom(new InternetAddress(sendMail)); 
159              message.setRecipients(Message.RecipientType.TO, 
160                      InternetAddress.parse(destMail)); 
161              message.setSubject(subjectMail); 
162              message.setText(contentMail); 
163              Transport.send(message); 
164              JOptionPane.showMessageDialog(null, "E-Mail sent!!"); 
165          } catch (MessagingException e) { 
166              throw new RuntimeException(e); 
167          } 
168   
169      }//GEN-LAST:event_jButton1ActionPerformed

171      private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton2ActionPerformed 
172          try { 
173              // TODO add your handling code here: 
174              Store store = RECEIVE_SESSION.getStore("imap"); 
175              store.connect(USERNAME, PASSWORD); 
176               
177              Folder inbox = store.getFolder("INBOX"); 
178              inbox.open(Folder.READ_ONLY); 
179              int messageCount = inbox.getMessageCount(); 
180    
181              jTextArea2.append("Total Messages:- " + messageCount); 
182    
183              Message[] messages = inbox.getMessages(); 
184              jTextArea2.append("------------------------------"); 
185    
186              for (int i = 0; i < 10; i++) { 
187                  jTextArea2.append("Mail Subject:- " + messages[i].getSubject()); 
188              } 
189    
190              inbox.close(true); 
191              store.close(); 
192          } catch (NoSuchProviderException ex) { 
193              Logger.getLogger(NewJFrame.class.getName()).log(Level.SEVERE, null, ex); 
194          } catch (MessagingException ex) { 
195              Logger.getLogger(NewJFrame.class.getName()).log(Level.SEVERE, null, ex); 
196          }
